pipeline {
   agent any
   tools {
       maven 'mvn'
       jdk 'jdk8'
   }

   stages {
      stage('Checkout'){
          steps{
              checkout([$class: 'GitSCM', branches: [[name: '*/master']], doGenerateSubmoduleConfigurations: false, extensions: [], submoduleCfg: [], userRemoteConfigs: [[url: 'https://github.com/SKumarSharma/spring-petclinic']]])
          }
      } 
       
      stage('Build') {
         steps {
            bat 'mvn clean package'
         }
      }
   }
}
